Pseudocode:

READ STORE Tahun;

        THEN IF Tahun mod 4 = 0 dan Tahun mod 100 = 0 dan Tahun mod 400 = 0;
        Hasil "KABISAT";
             ELSE Hasil "BUKAN KABISAT";
